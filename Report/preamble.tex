% === Custom Commands === %
% Count the number of characters in the given file including spaces
% Usage: \charcountspaces{<filename>}
% Example: \charcountspaces{main}
\newcommand{\charcountspaces}[1]{
\immediate\write18{expr `texcount -1 -sum -merge #1.tex` + `texcount -1 -sum -merge -char #1.tex` - 1 > charspaces.txt
}\input{charspaces.txt}}

% Count the number of characters in the given file excluding spaces
% Usage: \charcountspaces{<filename>}
% Example: \charcountspaces{main}
\newcommand{\charcount}[1]{
\immediate\write18{expr `texcount -1 -sum -merge -char #1.tex` - 1 > chars.txt
}\input{chars.txt}}

\newcommand{\wordcount}[1]{
\immediate\write18{expr `texcount -1 -inc -incbib -sum #1.tex` -1 > wordcount.txt}
\input{wordcount.txt}}
